\begin{algorithmic}
\Function{LCS-Length}{X, Y, m, n}
\State let $b[1\ldots m,\,1\ldots n]$ and $c[0\ldots m,\,0\ldots n]$ be new tables
\For{$i = 1$ \textbf{to} $m$}
  \State $c[i,0] \gets 0$
\EndFor
\For{$j = 0$ \textbf{to} $n$}
  \State $c[0,j] \gets 0$
\EndFor
\For{$i = 1$ \textbf{to} $m$}
  \For{$j = 1$ \textbf{to} $n$}
    \If{$x_i = y_j$}
      \State $c[i,j] \gets c[i-1,j-1] + 1$
      \State $b[i,j] \gets \nwarrow$
    \ElsIf{$c[i-1,j] \ge c[i,j-1]$}
      \State $c[i,j] \gets c[i-1,j]$
      \State $b[i,j] \gets \uparrow$
    \Else
      \State $c[i,j] \gets c[i,j-1]$
      \State $b[i,j] \gets \leftarrow$
    \EndIf
  \EndFor
\EndFor
\State \Return{$c, b$}
\EndFunction
\end{algorithmic}
